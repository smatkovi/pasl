# plotCps_corrected.txt - Korrigiertes Gnuplot für Zylinder

# Zuerst Anzahl der Zeilen bestimmen
stats "cps.txt" nooutput
n_lines = STATS_records

# Dateiname mit Anzahl der Panels erstellen
output_filename = sprintf("cps%d.png", n_lines)
set output output_filename

# Terminal-Einstellungen - gleiche Größe wie MATLAB
set terminal png size 1024,768 enhanced font "Arial,11"

# Linien-Stile
set style line 1 lc rgb '#FF8000' lw 2 pt 2 ps 1.3   # Orange mit x-Markern
set style line 2 lc rgb '#8000FF' lw 2.5              # Violett, nur Linie

# Beschriftungen
set title sprintf("Druckbeiwert - %d Panels", n_lines) font "Arial Bold,14"
set xlabel "Panel" font "Arial,12"
set ylabel "c_p" font "Arial,12"

# Gitter
set grid

# Legende 
set key inside right top box

# Achsenbereiche - WICHTIG
set xrange [0:1]
set yrange [-3:1]

# Achsen-Ticks
set xtics 0,0.1,1
set ytics -3,0.5,1

# WICHTIG: Für Zylinder - schließe die Kurve
# Plotte von 0 bis 1 (nicht nur bis (n-1)/n)
# Verwende parametrischen Modus für bessere Kontrolle

# Adaptive Darstellung
    # Mit Markern für wenige Panels
    plot "cps.txt" using ($0/(n_lines-0.999)):1 with linespoints ls 1 title 'Numerisch', \
         "cps.txt" using ($0/(n_lines-0.999)):2 with lines ls 2 title 'Exakt', \
         "cps.txt" using (1):1 every ::0::0 with points ls 1 notitle, \
         "cps.txt" using (1):2 every ::0::0 with lines ls 2 notitle


print sprintf("Plot erstellt: %s", output_filename)
