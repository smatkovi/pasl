# plotCps_universal.txt - Universelle Version die Zeilenanzahl automatisch erkennt

# Zuerst Anzahl der Zeilen bestimmen
stats "cps.txt" nooutput
n_lines = STATS_records

# Dateiname mit Anzahl der Panels erstellen
output_filename = sprintf("cps%d.png", n_lines)
set output output_filename

set terminal png size 1024,768

# Stil-Definitionen
set style data linespoints
set style line 1 lc rgb '#FF8800' lw 1.5 pt 2 ps 1.2  # Orange mit x-Markern (pt 2)
set style line 2 lc rgb '#8000FF' lw 2 pt 0           # Violett ohne Marker

# Beschriftungen
set title sprintf("Druckbeiwert - %d Panels", n_lines) font ",14"
set xlabel "Panel (normalisiert)" font ",12"
set ylabel "c_p" font ",12"

# Gitter und Legende
set grid
set key bottom center box

# Achsenbereiche
set xrange [0:1]
set yrange [-3:1]

# X-Ticks bei wichtigen Positionen
set xtics 0,0.1,1

# Plot mit normalisierter Panel-Position (0 bis 1)
# column(0) ist Zeilennummer (0-basiert), geteilt durch Gesamtzahl ergibt 0 bis ~0.875 für 8 Zeilen
# Daher verwenden wir column(0)/(n_lines-1) für exakt 0 bis 1, oder column(0)/n_lines für 0 bis (n-1)/n
plot "cps.txt" using (column(0)/n_lines):1 with linespoints ls 1 title 'c_p numerisch', \
     "cps.txt" using (column(0)/n_lines):2 with lines ls 2 title 'c_p analytisch'

# Debug-Ausgabe
print sprintf("Detected %d lines in file", n_lines)
print sprintf("Panel positions from 0 to %.3f", (n_lines-1.0)/n_lines)
print sprintf("Output saved as: %s", output_filename)
