# plotCps_universal.txt - Universelle Version die Zeilenanzahl automatisch erkennt

# Zuerst Anzahl der Zeilen bestimmen
stats "cps.txt" nooutput
n_lines = STATS_records

# Dateiname mit Anzahl der Panels erstellen
output_filename = sprintf("cps%d.png", n_lines)
set output output_filename

set terminal png size 1024,768

# Stil-Definitionen
set style data linespoints
set style line 1 lc rgb '#FF8000' lw 1.5 pt 7 ps 1.2  # Orange
set style line 2 lc rgb '#8000FF' lw 2 pt 0           # Violett

# Beschriftungen
set title sprintf("Druckbeiwert - %d Panels", n_lines) font ",14"
set xlabel "φ [°]" font ",12"
set ylabel "c_p" font ",12"

# Gitter und Legende
set grid
set key bottom center box

# Achsenbereiche
set xrange [0:360]
set yrange [-3:1]

# X-Ticks 
set xtics 0,45,360

# Plot mit automatischer Winkelberechnung
# column(0) ist wie $0 (Zeilennummer, 0-basiert)
plot "cps.txt" using (column(0)*360.0/n_lines):1 with linespoints ls 1 title 'c_p numerisch', \
     "cps.txt" using (column(0)*360.0/n_lines):2 with lines ls 2 title 'c_p analytisch'

# Debug-Ausgabe
print sprintf("Detected %d lines in file", n_lines)
print sprintf("Angle increment: %.1f degrees", 360.0/n_lines)
print sprintf("Output saved as: %s", output_filename)
